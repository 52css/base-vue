BaseLabel 标签 ​

使用场景 ​

  • 表单的标签
  • 表头的标签

默认注入tdesign组件, 也可以注入成其他UI组件库 ​

ts
import { type Component, ref } from 'vue';
import { Tooltip } from 'tdesign-vue-next';
import { HelpCircleIcon } from 'tdesign-icons-vue-next';
export const componentMap: Record<string, Component> = {
  Tooltip,
  HelpCircleIcon,
};
export const registerBaseLabelComponent = (
  name: string,
  component: Component
) => {
  componentMap[name] = component;
};

基本使用
<script lang="ts">
export interface BaseLabelDemoProps {
  prop1?: string;
}
export const BaseLabelDemoDefault = {};
export interface BaseLabelDemoEmits {
  (event: 'event1'): void;
}
</script>
<script setup lang="ts">
withDefaults(defineProps<BaseLabelDemoProps>(), BaseLabelDemoDefault);
defineEmits<BaseLabelDemoEmits>();
defineOptions({
  name: 'BaseLabelDemo',
});
</script>

<template>
  <h3>普通标签</h3>
  <base-label>普通标签</base-label>
  <h3>带全标签</h3>
  <base-label sub-label="副标签" tips="提示文案">普通标签</base-label>
</template>

<style scoped lang="scss">
.base-label-demo {
}
</style>
